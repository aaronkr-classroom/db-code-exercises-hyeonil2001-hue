create table club (
	id bigserial primary key,
	name varchar(50) not null,
	kind varchar(30)
)

insert into club(name, kind)
values
	('테니스', '운동'),
	('코딩', '공부'),
	('야구', '운동')

select * from club where kind='운동'

create table club_member (
	id bigserial primary key,
	name varchar(50) not null,
	age int not null,
	club_id foreign key references club(id),
	is_club_president boolean not null
)

insert into club_member(name, age, club_id, is_club_president)
values
	('kim1', 20, '1', true),
	('eom2', 30, '2', false),
	('jong3', 99, '3', true)

select * from club where is_club_president=true
